import random def generate_number(): numbers = list("0123456789") random.shuffle(numbers) return numbers[:3] def check(guess, answer): hit = 0 blow = 0 for i in range(3): if guess[i] == answer[i]: hit += 1 elif guess[i] in answer: blow += 1 return hit, blow def game(): answer = generate_number() attempts = 0 print("=== 数字推理ゲーム ===") print("3桁の数字を当ててね！（例：123）") while True: guess = list(input("入力：")) if len(guess) != 3 or not all(g.isdigit() for g in guess): print("3桁の数字を入力してね！") continue attempts += 1 hit, blow = check(guess, answer) print(f"{hit} Hit, {blow} Blow") if hit == 3: print(f"正解！{attempts}回でクリア！🎉") break game()
